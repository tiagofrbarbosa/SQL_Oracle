SELECT SUM(TOTAL) FROM(
SELECT P.CD_GRUPO_PROCEDIMENTO,
       P.CD_SUB_GRUPO_PROCEDIMENTO,
       E.CD_PROCEDIMENTO,
       P.DS_PROCEDIMENTO,
       COUNT(*) TOTAL
FROM EVE_SIASUS E,
     PROCEDIMENTO_SUS P
WHERE DT_EVE_SIASUS BETWEEN '04-08-2015' AND '05-08-2015'
AND E.CD_PROCEDIMENTO = P.CD_PROCEDIMENTO
AND E.CD_MULTI_EMPRESA = 1
GROUP BY P.CD_GRUPO_PROCEDIMENTO,P.CD_SUB_GRUPO_PROCEDIMENTO, E.CD_PROCEDIMENTO, P.DS_PROCEDIMENTO
ORDER BY 1,2,3
)
