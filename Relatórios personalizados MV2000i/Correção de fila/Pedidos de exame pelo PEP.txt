SELECT A.CD_ATENDIMENTO,
       TO_CHAR(A.DT_ATENDIMENTO,'DD/MM/YYYY') HORA,
       P.CD_PACIENTE,
       P.NM_PACIENTE,
       PR.NM_PRESTADOR,
       TIP.DS_TIP_PRESC
FROM ATENDIME A,
     PACIENTE P,
     PRESTADOR PR,
     PRE_MED PM,
     ITPRE_MED IT,
     TIP_PRESC TIP
WHERE A.CD_PACIENTE = P.CD_PACIENTE
AND PR.CD_PRESTADOR = A.CD_PRESTADOR
AND PM.CD_PRE_MED = IT.CD_PRE_MED
AND IT.CD_TIP_PRESC = TIP.CD_TIP_PRESC
AND PM.CD_ATENDIMENTO = A.CD_ATENDIMENTO
AND A.DT_ATENDIMENTO BETWEEN '01-07-2015' AND '02-05-2016'
AND A.CD_MULTI_EMPRESA = 1

SELECT TP.*,  TP.DS_TIP_PRESC FROM PRE_MED P, ITPRE_MED IT, TIP_PRESC TP WHERE IT.CD_PRE_MED = P.CD_PRE_MED AND IT.CD_TIP_PRESC = TP.CD_TIP_PRESC AND P.CD_ATENDIMENTO = 2736072

SELECT * FROM PRE_MED
SELECT * FROM ITPRE_MED
SELECT * FROM TIP_PRESC
