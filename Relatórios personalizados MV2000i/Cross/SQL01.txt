SELECT ACH.ID_PROFISSIONAL, P.NOME_PROFISSIONAL,
		ACH.ID_ESPECIALIDADE AS ID_ESPEC_ASSOC, E.NOME_ESPECIALIDADE AS NOME
		FROM age_consulta_hor AS ACH
		INNER JOIN especialidade AS E
		ON ACH.ID_ESPECIALIDADE=E.ID_ESPECIALIDADE
		INNER JOIN profissional AS P
		ON P.ID_PROFISSIONAL=ACH.ID_PROFISSIONAL
		WHERE ACH.COD_UNIDADE_EXECUTANTE = 9
		AND ACH.DATA_AGENDA = '2015-01-22'
		GROUP BY ACH.ID_PROFISSIONAL, P.NOME_PROFISSIONAL,
		ACH.ID_ESPECIALIDADE, E.NOME_ESPECIALIDADE
		ORDER BY E.NOME_ESPECIALIDADE
