SELECT REPLACE(WM_CONCAT(phone),',','-') phone, name, last_name, ID_Number
FROM(
SELECT '0' || CP.NR_DDD || CP.NR_TELEFONE phone,
       P.NM_PACIENTE name,
       null last_name,
       P.CD_PACIENTE ID_Number
FROM CONTATO_PACIENTE CP, 
     PACIENTE P,
     AGENDA_CENTRAL AC,
     IT_AGENDA_CENTRAL IT
WHERE P.CD_PACIENTE = CP.CD_PACIENTE(+)
AND AC.CD_AGENDA_CENTRAL = IT.CD_AGENDA_CENTRAL
AND IT.CD_PACIENTE = P.CD_PACIENTE
AND AC.DT_AGENDA = '21-07-2016'
AND AC.CD_PRESTADOR = 2611
ORDER BY DECODE(CP.TP_CONTATO,'C',1,'R',2,'O',3)
)
GROUP by name, last_name, ID_Number
