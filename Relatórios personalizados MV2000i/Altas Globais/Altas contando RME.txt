SELECT A.CD_ATENDIMENTO ATENDIMENTO,
       TO_CHAR(A.DT_ATENDIMENTO,'DD/MM/YYYY') DATA_ALTA,
       P.NM_PRESTADOR PRESTADOR,
       E.DS_ESPECIALID,
       A.CD_TIP_RES,
       (SELECT DS_TIP_RES FROM TIP_RES WHERE CD_TIP_RES = A.CD_TIP_RES) ALTA, 
       A.CD_CID,
       (SELECT DS_CID FROM CID WHERE CD_CID = A.CD_CID) DS_CID,
       (SELECT COUNT(VRD.REGISTRO_DOCUMENTO) FROM VDIC_RESPOSTA_DOCUMENTO VRD WHERE VRD.DOCUMENTO_PRONTUARIO = 164 AND VRD.ATENDIMENTO = A.CD_ATENDIMENTO) RME
FROM ATENDIME A,
     PRESTADOR P,
     ESPECIALID E
WHERE A.CD_MULTI_EMPRESA = 1
AND A.DT_ATENDIMENTO BETWEEN '01-09-2015' AND '30-09-2015'
AND A.CD_PRESTADOR = P.CD_PRESTADOR
AND A.CD_ESPECIALID = E.CD_ESPECIALID
AND A.TP_ATENDIMENTO = 'A'
ORDER BY A.CD_ATENDIMENTO
