SELECT ITEM.DS_ITEM_AGENDAMENTO, 
       A.CD_CID,
       COUNT(*) TOTAL
FROM ATENDIME A,
     IT_AGENDA_CENTRAL IT,
     ITEM_AGENDAMENTO ITEM
WHERE A.DT_ATENDIMENTO BETWEEN '01-08-2015' AND '15-08-2015'
AND A.CD_ATENDIMENTO = IT.CD_ATENDIMENTO
AND ITEM.CD_ITEM_AGENDAMENTO = IT.CD_ITEM_AGENDAMENTO
AND IT.CD_ITEM_AGENDAMENTO IN(1569)
AND A.CD_MULTI_EMPRESA = 1
GROUP BY ITEM.DS_ITEM_AGENDAMENTO, A.CD_CID
ORDER BY TOTAL DESC


SELECT ITEM.DS_ITEM_AGENDAMENTO, 
       A.CD_CID,
       SER.DS_SER_DIS
       --COUNT(*) TOTAL
FROM ATENDIME A,
     IT_AGENDA_CENTRAL IT,
     ITEM_AGENDAMENTO ITEM,
     SER_DIS SER
WHERE A.DT_ATENDIMENTO BETWEEN '01-08-2015' AND '15-08-2015'
AND A.CD_ATENDIMENTO = IT.CD_ATENDIMENTO
AND ITEM.CD_ITEM_AGENDAMENTO = IT.CD_ITEM_AGENDAMENTO
AND IT.CD_ITEM_AGENDAMENTO IN(1569)
AND A.CD_MULTI_EMPRESA = 1
AND SER.CD_SER_DIS = IT.CD_SER_DIS
GROUP BY ITEM.DS_ITEM_AGENDAMENTO, A.CD_CID
ORDER BY TOTAL DESC
