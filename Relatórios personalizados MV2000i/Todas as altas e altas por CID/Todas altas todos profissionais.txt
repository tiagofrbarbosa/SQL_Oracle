SELECT A.CD_ATENDIMENTO "ATENDIMENTO",
       P.CD_PACIENTE,
       P.NM_PACIENTE "NOME",
       TO_CHAR(P.DT_NASCIMENTO,'DD/MM/YYYY') "NASCIMENTO",
       trunc((months_between(sysdate, P.DT_NASCIMENTO))/12) "IDADE",
       P.TP_SEXO "SEXO",
       TO_CHAR(A.DT_ATENDIMENTO,'DD/MM/YYYY') "DATA ATENDIMENTO",
       E.DS_ESPECIALID "ESPECIALIDADE",
       PRE.CD_PRESTADOR,
       PRE.NM_PRESTADOR,
       (SELECT CID.CD_CID FROM CID WHERE A.CD_CID = CID.CD_CID) CD_CID,
        (SELECT CID.DS_CID FROM CID WHERE A.CD_CID = CID.CD_CID) DS_CID,
       (SELECT DS_TIP_RES FROM TIP_RES WHERE TIP_RES.CD_TIP_RES = A.CD_TIP_RES) ALTA,
       (SELECT COUNT(VRD.REGISTRO_DOCUMENTO) FROM VDIC_RESPOSTA_DOCUMENTO VRD WHERE VRD.DOCUMENTO_PRONTUARIO = 164 AND VRD.ATENDIMENTO = A.CD_ATENDIMENTO) RME
FROM ATENDIME A,
     PACIENTE P,
     ESPECIALID E,
     PRESTADOR PRE
WHERE A.DT_ATENDIMENTO BETWEEN '01-01-2015' AND '31-12-2015'
AND P.CD_PACIENTE = A.CD_PACIENTE
AND A.CD_ESPECIALID = E.CD_ESPECIALID
AND A.CD_MULTI_EMPRESA = 1
AND A.TP_ATENDIMENTO = 'A'
AND PRE.CD_PRESTADOR = A.CD_PRESTADOR
ORDER BY 1
