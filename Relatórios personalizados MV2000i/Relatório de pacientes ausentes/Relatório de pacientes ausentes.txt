SELECT TO_CHAR(AC.DT_AGENDA,'DD/MM/YYYY')DATA_AGENDA,
       IT.CD_ATENDIMENTO,
       IT.CD_PACIENTE,
       IT.NM_PACIENTE,
       P.NM_PRESTADOR,
       DECODE((SELECT DS_TIP_MAR FROM TIP_MAR WHERE CD_TIP_MAR = IT.CD_TIP_MAR),null,'EXAME',(SELECT DS_TIP_MAR FROM TIP_MAR WHERE CD_TIP_MAR = IT.CD_TIP_MAR)) TIPO,
       ITEM.DS_ITEM_AGENDAMENTO
      FROM IT_AGENDA_CENTRAL IT,
           AGENDA_CENTRAL AC,
           PRESTADOR P,
           ITEM_AGENDAMENTO ITEM
  WHERE IT.CD_AGENDA_CENTRAL = AC.CD_AGENDA_CENTRAL
        AND ITEM.CD_ITEM_AGENDAMENTO = IT.CD_ITEM_AGENDAMENTO
        AND IT.NM_PACIENTE IN(
                    SELECT IT.NM_PACIENTE
                    FROM 
                    IT_AGENDA_CENTRAL IT,
                    AGENDA_CENTRAL AG,
                    TIP_MAR TP,
                    PRESTADOR P
                    WHERE 
                    --AG.CD_PRESTADOR = 2768
                    AG.DT_AGENDA BETWEEN '21-03-2016' AND '21-03-2016'
                    AND AG.CD_AGENDA_CENTRAL = IT.CD_AGENDA_CENTRAL
                    AND IT.CD_TIP_MAR = TP.CD_TIP_MAR
                    AND P.CD_PRESTADOR = AG.CD_PRESTADOR
                    AND IT.CD_ATENDIMENTO IS NULL
                    )
AND AC.DT_AGENDA BETWEEN '21-03-2016' AND '21-03-2017'
AND P.CD_PRESTADOR = AC.CD_PRESTADOR
AND IT.CD_ATENDIMENTO IS NULL
ORDER BY IT.NM_PACIENTE ASC
