SELECT A.CD_ATENDIMENTO,
       A.DT_ATENDIMENTO,
       A.CD_PACIENTE,
       P.NM_PACIENTE,
       INFO.DS_INFO_ATENDIMENTO,
       A.NM_USUARIO
FROM ATENDIME A,
     PACIENTE P,
     ATENDIME_INFO INFO
WHERE A.CD_PACIENTE = P.CD_PACIENTE
AND A.CD_MULTI_EMPRESA = 1
AND A.DT_ATENDIMENTO BETWEEN '01-12-2015' AND '01-03-2016'
AND INFO.CD_ATENDIMENTO = A.CD_ATENDIMENTO(+)
ORDER BY A.CD_ATENDIMENTO
