SELECT A.CD_ATENDIMENTO,
       A.CD_PACIENTE,
       P.NM_PACIENTE,
        VDOC.DESC_DOCUMENTO_PRONTUARIO,
        VIRD.DESC_RESPOSTA
FROM VDIC_RESPOSTA_DOCUMENTO VRD,
     VDIC_ITEM_RESPOSTA_DOCUMENTO  VIRD,
     VDIC_DOCUMENTO_PRONTUARIO VDOC,
     PACIENTE P,
     ATENDIME A
WHERE VRD.DOCUMENTO_PRONTUARIO IN(111,294)
     AND A.CD_ATENDIMENTO = VRD.ATENDIMENTO
     AND A.CD_PACIENTE = P.CD_PACIENTE
     AND A.CD_MULTI_EMPRESA = 1
     AND A.DT_ATENDIMENTO BETWEEN '01-01-2016' AND '20-01-2016'
     AND VIRD.REGISTRO_DOCUMENTO = VRD.REGISTRO_DOCUMENTO
     AND VRD.DOCUMENTO_PRONTUARIO = VDOC.DOCUMENTO_PRONTUARIO
     AND VIRD.PERGUNTA_DOCUMENTO IN(6532,6535,6538,6541,6544)
AND UPPER(VIRD.DESC_RESPOSTA) LIKE '%BIO%'

UNION ALL
