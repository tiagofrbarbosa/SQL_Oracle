SELECT IT.CD_ITSOLICITACAO_OS CD_OS,
       TO_CHAR(IT.HR_INICIO,'DD/MM/YYYY HH24:MI') INICIO,
       TO_CHAR(IT.HR_FINAL,'DD/MM/YYYY HH24:MI') FIM,
       IT.CD_FUNC CD_FUNC,
       F.NM_FUNC NOME,
       S.NM_SETOR SETOR,
       IT.VL_TEMPO_GASTO_MIN TEMPO_GASTO
FROM ITSOLICITACAO_OS IT, FUNCIONARIO F, SOLICITACAO_OS OS, SETOR S
WHERE IT.HR_INICIO BETWEEN '21-03-2014' AND '14-04-2014'
AND IT.CD_FUNC = F.CD_FUNC
AND IT.CD_OS = OS.CD_OS
AND OS.CD_MULTI_EMPRESA = 1
AND OS.CD_OFICINA = 5
AND OS.CD_SETOR = S.CD_SETOR
ORDER BY IT.CD_ITSOLICITACAO_OS
