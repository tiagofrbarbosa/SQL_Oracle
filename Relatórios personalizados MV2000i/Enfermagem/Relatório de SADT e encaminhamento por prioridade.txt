SELECT A.CD_ATENDIMENTO,
       A.CD_PACIENTE,
       P.NM_PACIENTE,
       (SELECT CON.NR_DDD || '-' || CON.NR_TELEFONE FROM CONTATO_PACIENTE CON WHERE CON.CD_PACIENTE = A.CD_PACIENTE AND CON.TP_CONTATO = 'R' AND CON.SN_PADRAO = 'S' AND ROWNUM = 1) || '   ' || (SELECT CON.NR_DDD || '-' || CON.NR_TELEFONE FROM CONTATO_PACIENTE CON WHERE CON.CD_PACIENTE = A.CD_PACIENTE AND CON.TP_CONTATO = 'C' AND CON.SN_PADRAO = 'S' AND ROWNUM = 1) || '   ' || (SELECT CON.NR_DDD || '-' || CON.NR_TELEFONE FROM CONTATO_PACIENTE CON WHERE CON.CD_PACIENTE = A.CD_PACIENTE AND CON.TP_CONTATO = 'O' AND CON.SN_PADRAO = 'S' AND ROWNUM = 1) nr_fone,
        TO_CHAR(A.DT_ATENDIMENTO,'DD/MM/YYYY') DATA,
        PRE.CD_PRESTADOR,
        PRE.NM_PRESTADOR,
        VDOC.DESC_DOCUMENTO_PRONTUARIO DOC,
        VIRD.DESC_RESPOSTA,
        (SELECT DESC_RESPOSTA FROM VDIC_ITEM_RESPOSTA_DOCUMENTO WHERE REGISTRO_DOCUMENTO = VRD.REGISTRO_DOCUMENTO AND PERGUNTA_DOCUMENTO = 34940)
FROM VDIC_RESPOSTA_DOCUMENTO VRD,
     VDIC_ITEM_RESPOSTA_DOCUMENTO  VIRD,
     VDIC_DOCUMENTO_PRONTUARIO VDOC,
     PACIENTE P,
     ATENDIME A,
     PRESTADOR PRE
WHERE VRD.DOCUMENTO_PRONTUARIO IN(111,121)
     AND A.CD_ATENDIMENTO = VRD.ATENDIMENTO
     AND A.CD_PACIENTE = P.CD_PACIENTE
     AND A.CD_MULTI_EMPRESA = 1
     AND TO_DATE(A.DT_ATENDIMENTO,'DD/MM/YYYY') BETWEEN TO_DATE(:DT_INICIAL,'DD/MM/YYYY') AND TO_DATE(:DT_FINAL,'DD/MM/YYYY')
     --AND A.DT_ATENDIMENTO BETWEEN '12-02-2016' AND '13-02-2016'
     AND VIRD.REGISTRO_DOCUMENTO = VRD.REGISTRO_DOCUMENTO
     AND VRD.DOCUMENTO_PRONTUARIO = VDOC.DOCUMENTO_PRONTUARIO
     AND VIRD.PERGUNTA_DOCUMENTO IN(6532,6772)
     AND PRE.CD_PRESTADOR = A.CD_PRESTADOR
