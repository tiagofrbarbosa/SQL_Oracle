SELECT NM_SETOR,
       NM_SET_EXA,
       SUM(TOTAL)
FROM(
SELECT PR.DT_PEDIDO,
       PR.CD_PED_RX,
       PR.CD_SETOR,
       S.NM_SETOR,
       SE.CD_SET_EXA,
       SE.NM_SET_EXA,COUNT(*) TOTAL
FROM PED_RX PR, SETOR S, ATENDIME A, SET_EXA SE, IT_AGENDA_CENTRAL IT,ITEM_AGENDAMENTO IA
WHERE PR.CD_SETOR = S.CD_SETOR
AND PR.CD_ATENDIMENTO = A.CD_ATENDIMENTO
AND IT.CD_ATENDIMENTO = A.CD_ATENDIMENTO
AND IA.CD_ITEM_AGENDAMENTO = IT.CD_ITEM_AGENDAMENTO
AND PR.CD_SET_EXA = SE.CD_SET_EXA
AND A.CD_MULTI_EMPRESA = 1
AND PR.DT_PEDIDO BETWEEN '16-03-2014' AND '17-03-2014'
GROUP BY PR.DT_PEDIDO,PR.CD_PED_RX,PR.CD_SETOR,S.NM_SETOR,SE.CD_SET_EXA,SE.NM_SET_EXA
ORDER BY NM_SET_EXA
)
GROUP BY NM_SETOR,NM_SET_EXA
