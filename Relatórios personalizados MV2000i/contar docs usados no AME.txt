SELECT DOCUMENTO_PRONTUARIO,
       SUM(TOTAL)
       FROM(
SELECT VRD.DOCUMENTO_PRONTUARIO, VRD.REGISTRO_DOCUMENTO, COUNT(*) TOTAL 
FROM VDIC_RESPOSTA_DOCUMENTO VRD,
     ATENDIME A
WHERE VRD.DOCUMENTO_PRONTUARIO IN(121,291,236,57,377,177,117,129,93,284,283,384,382,293,131)
AND VRD.ATENDIMENTO = A.CD_ATENDIMENTO
AND TO_CHAR(A.DT_ATENDIMENTO,'YYYY') = '2015'
AND A.CD_MULTI_EMPRESA = 1
GROUP BY VRD.DOCUMENTO_PRONTUARIO,VRD.REGISTRO_DOCUMENTO)
GROUP BY DOCUMENTO_PRONTUARIO
