SELECT IT.CD_ATENDIMENTO,
       IT.CD_PACIENTE,
       IT.NM_PACIENTE,
       IT.DT_NASCIMENTO,
       (SELECT NM_MAE FROM PACIENTE WHERE CD_PACIENTE = IT.CD_PACIENTE) MAE,
       TO_CHAR(AG.DT_AGENDA,'DD/MM/YYYY') DATA,
       TO_CHAR(IT.HR_AGENDA,'HH24:MI') HORA,
       TP.DS_TIP_MAR,
       P.NM_PRESTADOR
FROM 
IT_AGENDA_CENTRAL IT,
AGENDA_CENTRAL AG,
TIP_MAR TP,
PRESTADOR P
WHERE AG.CD_PRESTADOR = 2768
AND AG.CD_AGENDA_CENTRAL = IT.CD_AGENDA_CENTRAL
AND IT.CD_TIP_MAR = TP.CD_TIP_MAR
AND P.CD_PRESTADOR = AG.CD_PRESTADOR
ORDER BY AG.DT_AGENDA ASC
