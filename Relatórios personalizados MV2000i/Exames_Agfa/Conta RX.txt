SELECT IT.CD_PACIENTE, IT.NM_PACIENTE, ER.DS_EXA_RX, IT.CD_ATENDIMENTO, PR.CD_PED_RX
FROM IT_AGENDA_CENTRAL IT,
     AGENDA_CENTRAL AC,
     ITEM_AGENDAMENTO ITEM,
     PED_RX PR,
     ITPED_RX ITRX,
     EXA_RX ER
WHERE IT.CD_AGENDA_CENTRAL = AC.CD_AGENDA_CENTRAL
AND AC.DT_AGENDA = '19-07-2016'
AND AC.CD_PRESTADOR IN(85,187)
AND ITEM.CD_ITEM_AGENDAMENTO = IT.CD_ITEM_AGENDAMENTO
--AND ITEM.DS_ITEM_AGENDAMENTO LIKE 'RX%'
AND IT.CD_ATENDIMENTO IS NOT NULL
AND PR.CD_ATENDIMENTO = IT.CD_ATENDIMENTO
AND ITRX.CD_PED_RX = PR.CD_PED_RX
AND ITRX.CD_EXA_RX = ER.CD_EXA_RX
ORDER BY IT.NM_PACIENTE
