SELECT P.NM_PRESTADOR "Médico",
       AC.DT_AGENDA "Data",
       TO_CHAR(AC.HR_INICIO,'HH24:MI') "Hora Inicial",
       TO_CHAR(AC.HR_FIM,'HH24:MI') "Hora Final",
       AC.QT_ATENDIMENTO "Número Consultas",
       AC.QT_MARCADOS "Número Marcados",
       AC.QT_TEMPO_MEDIO "Tempo Médio",
       AC.SN_SIA "Atende SIA",
       RC.DS_RECURSO_CENTRAL "Serviço"
FROM PRESTADOR P, AGENDA_CENTRAL AC, RECURSO_CENTRAL RC
WHERE P.CD_PRESTADOR = AC.CD_PRESTADOR
AND RC.CD_RECURSO_CENTRAL = AC.CD_RECURSO_CENTRAL
AND AC.DT_AGENDA BETWEEN '30/09/2013' AND '30/09/2013';

SELECT DISTINCT(AC.CD_AGENDA_CENTRAL),
       P.NM_PRESTADOR,
       AC.DT_AGENDA,
       TO_CHAR(AC.HR_INICIO,'HH24:MI'),
       TO_CHAR(AC.HR_FIM,'HH24:MI'),

       AC.QT_MARCADOS,
       AC.QT_TEMPO_MEDIO,
       AC.SN_SIA,
       SD.DS_SER_DIS

INNER JOIN IT_AGENDA_CENTRAL IT ON AC.CD_AGENDA_CENTRAL = IT.CD_AGENDA_CENTRAL

INNER JOIN PRESTADOR P ON AC.CD_PRESTADOR = P.CD_PRESTADOR
ORDER BY AC.CD_AGENDA_CENTRAL,P.NM_PRESTADOR;
