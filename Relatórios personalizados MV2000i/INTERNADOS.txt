SELECT DISTINCT(A.CD_ATENDIMENTO) "ATENDIMENTO",
       A.CD_PACIENTE "CÓDIGO PACIENTE",
       P.NM_PACIENTE "PACIENTE",
       TO_CHAR(A.DT_ATENDIMENTO,'dd/mm/yyyy') "DATA ATENDIMENTO",
       TO_CHAR(A.DT_ALTA,'dd/mm/yyyy') "DATA ALTA",
       TO_DATE(A.DT_ALTA,'dd/mm/yyyy') - TO_DATE(A.DT_ATENDIMENTO,'dd/mm/yyyy') "DIA(S) INTERNADO",
       A.CD_CID "CÓDIGO CID",
       C.DS_CID "DS CID"
       FROM ATENDIME A,PACIENTE P, CID C, TIP_RES TIP
       WHERE A.CD_PACIENTE = P.CD_PACIENTE(+)
       AND A.CD_CID = C.CD_CID(+)
       AND A.TP_ATENDIMENTO = 'I'
       AND A.DT_ATENDIMENTO BETWEEN '01-JAN-2013' AND '31-AGO-2013'
       ORDER BY A.CD_ATENDIMENTO;
