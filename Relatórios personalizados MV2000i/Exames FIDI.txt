SELECT DISTINCT(RX.CD_EXA_RX),
       RX.DS_EXA_RX
FROM 
DBAMV.EXA_RX RX,
DBAMV.ITEM_AGENDAMENTO ITAGE,
DBAMV.IT_AGENDA_CENTRAL IT,
DBAmv.AGENDA_CENTRAL AC
WHERE RX.CD_EXA_RX = ITAGE.CD_EXA_RX
AND ITAGE.TP_ITEM = 'I'
AND ITAGE.CD_ITEM_AGENDAMENTO = IT.CD_ITEM_AGENDAMENTO
AND IT.CD_AGENDA_CENTRAL = AC.CD_AGENDA_CENTRAL
AND AC.DT_AGENDA BETWEEN '01-01-2014' AND '21-05-2015'
ORDER BY 1
