select cd_paciente
       ,nm_paciente
       ,cep_logradouros.cd_tipo_logradouro
       ,decode(tb_logradouros.log_tipo,'R','RUA','Av','AVENIDA','Est','ESTRADA','Tv','TRAVESSA','Acesso','Acesso',' ','VAZIO')TIPO
       ,cep_sus.cd_uf
       ,cidade.nm_cidade cidade
       ,tb_logradouros.log_bairro
       ,paciente.nr_cep
       ,ds_endereco
       ,PACIENTE.nr_endereco
       ,LENGTH(ds_endereco)TOTAL 
  from paciente,cep_logradouros, tb_logradouros, cep_sus, cidade 
 where cd_paciente in (select distinct(cd_paciente) from eve_siasus where cd_remessa = 769)
   and length(ds_endereco) >30
   and paciente.nr_cep = cep_logradouros.nr_cep
   and cep_logradouros.nr_cep = tb_logradouros.log_cep(+) 
   and cep_logradouros.nr_cep = cep_sus.nr_cep
   and paciente.cd_cidade = cidade.cd_cidade(+)
