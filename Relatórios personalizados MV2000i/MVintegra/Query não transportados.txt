SELECT i.CD_IMV_AGENDAMENTO,
       i.CD_PACIENTE_INTEGRA,
       i.CD_AGENDAMENTO_INTEGRA,
       i.TP_ACAO_AGENDA,
       i.DT_GERADO,
       l.NM_SERVICO,
       i.DS_ERRO,
       i.DT_AGENDA,
       i.DS_ESPECIALIDADE,
       i.NM_PRESTADOR,
       i.NM_PACIENTE,
       TO_CHAR(i.HR_INICIO_AGENDA,'HH:MM') HR_INI,
       TO_CHAR(i.HR_FINAL_AGENDA,'HH:MM') HR_FIN
FROM MVINTEGRA.IMV_AGENDAMENTO i,
     MVINTEGRA.IMV_LOG_REQUISICAO l
WHERE i.DT_GERADO BETWEEN '18-11-2014' AND '19-11-2014'
    AND i.CD_IMV_LOG_REQUISICAO = l.CD_IMV_LOG_REQUISICAO
    AND l.SN_SUCESSO = 'N'
    ORDER BY 1 asc
    
SELECT i.CD_IMV_AGENDAMENTO,
	   i.CD_AGENDAMENTO_INTEGRA,
       i.TP_ACAO_AGENDA,
       i.DT_GERADO,
       l.NM_SERVICO,
       i.DS_ERRO,
       i.DT_AGENDA,
       i.DS_ESPECIALIDADE,
       i.NM_PRESTADOR,
       i.NM_PACIENTE,
       TO_CHAR(i.HR_INICIO_AGENDA,'HH:MM') HR_INI,
       TO_CHAR(i.HR_FINAL_AGENDA,'HH:MM') HR_FIN
FROM MVINTEGRA.IMV_AGENDAMENTO i,
     MVINTEGRA.IMV_LOG_REQUISICAO l
WHERE i.CD_AGENDAMENTO_INTEGRA IN(
        SELECT i.CD_AGENDAMENTO_INTEGRA
        FROM MVINTEGRA.IMV_AGENDAMENTO i,
             MVINTEGRA.IMV_LOG_REQUISICAO l
        WHERE i.DT_GERADO BETWEEN '12-11-2014' AND '14-11-2014'
        AND i.CD_IMV_LOG_REQUISICAO = l.CD_IMV_LOG_REQUISICAO
        AND l.SN_SUCESSO = 'N')
        AND i.CD_IMV_LOG_REQUISICAO = l.CD_IMV_LOG_REQUISICAO
        ORDER BY i.CD_AGENDAMENTO_INTEGRA
