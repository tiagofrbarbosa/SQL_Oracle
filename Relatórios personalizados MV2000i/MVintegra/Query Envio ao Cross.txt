SELECT i.DT_AGENDA DT_AGENDA,
       i.CD_PACIENTE_INTEGRA CD_PACIENTE,
       i.NM_PACIENTE PACIENTE,
       TO_CHAR(i.HR_INICIO_AGENDA,'HH:MM') HORARIO
       l.NM_SERVICO TIPO,
       i.CD_AGENDAMENTO_INTEGRA CD_AGENDAMENTO,
       i.DS_ESPECIALIDADE ESPECIALIDADE,
       i.NM_PRESTADOR PRESTADOR,
       i.DT_GERADO,
       i.DS_ERRO ERRO
FROM MVINTEGRA.IMV_AGENDAMENTO i,
     MVINTEGRA.IMV_LOG_REQUISICAO l
WHERE i.DT_GERADO BETWEEN '18-11-2014' AND '19-11-2014'
    AND i.CD_IMV_LOG_REQUISICAO = l.CD_IMV_LOG_REQUISICAO
    AND l.SN_SUCESSO = 'N'
    ORDER BY 1 asc
