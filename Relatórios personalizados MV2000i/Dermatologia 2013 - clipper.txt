SELECT i.cd_atendimento,
       I.CD_PACIENTE,
       I.NM_PACIENTE,
       I.DT_NASCIMENTO,
       A.CD_CID,
       U.DS_UNIDADE_REGIONAL,
       S.DS_SER_DIS
from diretoria_regional d
                        ,it_marcacao i
                        ,unidade_regional u
                        ,marcacao m
                        ,ser_dis s
                        ,atendime a
                        where d.cd_diretoria_regional = u.cd_diretoria_regional
                        and u.cd_unidade_regional = i.cd_unidade_regional
                        and i.cd_marcacao = m.cd_marcacao
                        and i.cd_ser_dis = s.cd_ser_dis
                        and I.CD_ATENDIMENTO = A.CD_ATENDIMENTO
                        and m.dt_marcacao between '01-JAN-2013' and  '31-DEZ-2013'
                        and i.nm_paciente is not null
                        AND I.CD_SER_DIS = 16
                        AND I.CD_TIP_MAR IN('1','2')
                        AND U.CD_UNIDADE_REGIONAL IN('33022','118915');



SELECT A.CD_ATENDIMENTO,
       TO_CHAR(A.DT_ATENDIMENTO,'DD/MM/YYYY') DATA_ATENDIMENTO,
       A.CD_PACIENTE,
       P.NM_PACIENTE,
       TO_CHAR(P.DT_NASCIMENTO,'DD/MM/YYYY') DATA_NASCIMENTO,
       A.CD_CID,
       L.DS_LOC_PROCED,
       S.DS_SER_DIS
FROM ATENDIME A,PACIENTE P,SER_DIS S,LOC_PROCED L
      WHERE A.DT_ATENDIMENTO BETWEEN '01-09-2013' AND '30-09-2013'
      AND A.CD_PACIENTE = P.CD_PACIENTE
      AND A.CD_LOC_PROCED = L.CD_LOC_PROCED
      AND A.CD_SER_DIS = S.CD_SER_DIS
       and A.CD_paciente is not null 
                        AND A.CD_SER_DIS = 16 
                        AND A.CD_TIP_MAR = 1 
                        ORDER BY 1,2 ASC;
