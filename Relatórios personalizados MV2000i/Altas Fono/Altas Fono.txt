SELECT DISTINCT(A.CD_ATENDIMENTO),
       TO_CHAR(A.DT_ATENDIMENTO,'DD/MM/YYYY') DATA,
       P.CD_PACIENTE,
       P.NM_PACIENTE,
       (SELECT DS_TIP_MAR FROM TIP_MAR WHERE CD_TIP_MAR = A.CD_TIP_MAR) TIP_MAR,
       (SELECT DS_TIP_RES FROM TIP_RES WHERE CD_TIP_RES = A.CD_TIP_RES)TIP_RES,
       PRE.NM_PRESTADOR,
       ITEM.DS_ITEM_AGENDAMENTO,
       SER.DS_SER_DIS
FROM ATENDIME A,
     PACIENTE P,
     PRESTADOR PRE,
     IT_AGENDA_CENTRAL IT,
     ITEM_AGENDAMENTO ITEM,
     SER_DIS SER
WHERE A.CD_MULTI_EMPRESA = 1
AND A.CD_ESPECIALID = 54
AND A.DT_ATENDIMENTO BETWEEN '01-01-2015' AND '31-12-2015'
AND A.CD_PACIENTE = P.CD_PACIENTE
AND A.CD_PRESTADOR = PRE.CD_PRESTADOR
AND IT.CD_ATENDIMENTO = A.CD_ATENDIMENTO
AND ITEM.CD_ITEM_AGENDAMENTO = IT.CD_ITEM_AGENDAMENTO(+)
AND SER.CD_SER_DIS = IT.CD_SER_DIS
AND A.TP_ATENDIMENTO = 'A'
ORDER BY A.CD_ATENDIMENTO
