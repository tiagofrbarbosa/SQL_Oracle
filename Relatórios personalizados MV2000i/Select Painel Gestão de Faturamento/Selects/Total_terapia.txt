SELECT COUNT(ESP.DS_ESPECIALID) TOTAL
FROM ATENDIME ATE,
TIP_MAR MAR,
ESPECIALID ESP,
SER_DIS SER
WHERE DT_ATENDIMENTO BETWEEN '01/07/2015' AND '01/07/2015'
AND ATE.TP_ATENDIMENTO = 'A'
AND ATE.CD_MULTI_EMPRESA = 1
AND ATE.CD_ESPECIALID IN(54)
and MAR.CD_TIP_MAR = ATE.CD_TIP_MAR
AND ATE.CD_ESPECIALID = ESP.CD_ESPECIALID
AND ATE.CD_TIP_MAR = MAR.CD_TIP_MAR
AND ATE.CD_SER_DIS = SER.CD_SER_DIS
AND MAR.DS_TIP_MAR not IN('PRIMEIRA CONSULTA')
