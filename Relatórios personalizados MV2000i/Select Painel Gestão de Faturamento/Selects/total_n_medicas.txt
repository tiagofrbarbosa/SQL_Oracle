SELECT SUM(TOTAL) 
FROM(
SELECT COUNT(CD_ATENDIMENTO) TOTAL
FROM ATENDIME
WHERE TO_CHAR(DT_ATENDIMENTO, 'MM/YYYY') = TO_CHAR(sysdate, 'MM/YYYY')
AND CD_MULTI_EMPRESA = 1
AND TP_ATENDIMENTO = 'A'
AND CD_ESPECIALID IN(54,55,3,62)
AND CD_TIP_MAR IN(1)

UNION ALL

SELECT COUNT(CD_ATENDIMENTO) TOTAL
FROM ATENDIME
WHERE TO_CHAR(DT_ATENDIMENTO, 'MM/YYYY') = TO_CHAR(sysdate, 'MM/YYYY')
AND CD_MULTI_EMPRESA = 1
AND TP_ATENDIMENTO = 'A'
AND CD_ESPECIALID IN(62,3)
AND CD_TIP_MAR IN(2)


UNION ALL

SELECT COUNT(CD_ATENDIMENTO) TOTAL 
FROM ATENDIME
WHERE TO_CHAR(DT_ATENDIMENTO, 'MM/YYYY') = TO_CHAR(sysdate, 'MM/YYYY')
AND CD_MULTI_EMPRESA = 1
AND TP_ATENDIMENTO = 'A'
AND CD_ESPECIALID = 65

UNION ALL

SELECT COUNT(CD_ATENDIMENTO) TOTAL 
FROM ATENDIME
WHERE TO_CHAR(DT_ATENDIMENTO, 'MM/YYYY') = TO_CHAR(sysdate) 'MM/YYYY')
AND CD_MULTI_EMPRESA = 1
AND TP_ATENDIMENTO = 'A'
AND CD_ESPECIALID = 1
AND CD_PROCEDIMENTO = 0301010030
)
