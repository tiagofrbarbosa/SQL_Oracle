SELECT I.CD_PACIENTE_INTEGRA,
       I.NM_PACIENTE,
       I.DT_NASCIMENTO,
       I.NM_MAE,
       I.DT_GERADO,
       I.DT_AGENDA,
       TO_CHAR(I.HR_INICIO_AGENDA,'HH24:MI') HORA,
       I.NM_PRESTADOR,
       I.DS_ESPECIALIDADE
FROM MVINTEGRA.IMV_AGENDAMENTO I
WHERE I.DT_GERADO BETWEEN '08-09-2015' AND '28-12-2015'
AND I.TP_STATUS = 'T'
AND I.TP_AGENDA = 'AMBU'
AND I.TP_MOVIMENTO IN('I','A')
AND I.NM_PACIENTE IS NOT NULL
